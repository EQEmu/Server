CMAKE_MINIMUM_REQUIRED(VERSION 3.12)

SET(shared_memory_sources
	items.cpp
	main.cpp
	spells.cpp
	skill_caps.cpp
)

	items.h
	spells.h

SET(EXECUTABLE_OUTPUT_PATH ${PROJECT_BINARY_DIR}/bin)
